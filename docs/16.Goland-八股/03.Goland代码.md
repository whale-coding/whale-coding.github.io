## 代码面试题

### 1.三个goroutinue交替打印abc 10次

```go
package main

import (
	"fmt"
	"sync"
)

func main() {
	const times = 10
	var wg sync.WaitGroup

	// 创建三个通道，用于控制流转
	// chanA 控制打印 'a' 的权限
	// chanB 控制打印 'b' 的权限
	// chanC 控制打印 'c' 的权限
(chanA := make(chan struct{}, 1)
(chanB := make(chan struct{}, 1)
(chanC := make(chan struct{}, 1)

	// 初始时，只给 chanA 放入一个令牌，让 'a' 先打印
	chanA <- struct{}{}

	wg.Add(3)

	// Goroutine A: 打印 'a'
	go func() {
		defer wg.Done()
		for i := 0; i < times; i++ {
			<-chanA // 等待接收令牌
			fmt.Print("a")
			chanB <- struct{}{} // 将令牌传递给 B
		}
	}()

	// Goroutine B: 打印 'b'
	go func() {
		defer wg.Done()
		for i := 0; i < times; i++ {
			<-chanB // 等待接收令牌
			fmt.Print("b")
			chanC <- struct{}{} // 将令牌传递给 C
		}
	}()

	// Goroutine C: 打印 'c'
	go func() {
		defer wg.Done()
		for i := 0; i < times; i++ {
			<-chanC // 等待接收令牌
			fmt.Print("c")
			if i == times-1 {
				// 最后一次打印完成后，可以选择关闭通道或不做处理
				// 这里为了简单，不关闭，依靠主程序退出
				break
			}
			chanA <- struct{}{} // 将令牌传递回 A
		}
	}()

	wg.Wait()
	fmt.Println() // 换行
}
 
 
 代码逻辑解析：
创建三个无缓冲（或缓冲为1）的通道 chanA, chanB, chanC。
初始化时，向 chanA 发送一个信号，允许第一个 goroutine 执行。
Goroutine A 接收到信号后打印 "a"，然后向 chanB 发送信号，唤醒 B。
Goroutine B 接收到信号后打印 "b"，然后向 chanC 发送信号，唤醒 C。
Goroutine C 接收到信号后打印 "c"，然后向 chanA 发送信号，完成一轮循环，再次唤醒 A。
循环 10 次后，所有 goroutine 结束，主程序等待所有任务完成 (wg.Wait())。
 
 
 方案二：使用互斥锁和条件变量 (Mutex + Cond)
```



### 2.两个goroutinue交替打印奇偶数

```go
package main



import "fmt"



func main() {

    // 定义无缓冲通道，用于协程间同步

    ch := make(chan struct{})



    // 奇数协程：打印1,3,5...99

    go func() {

        for i := 1; i <= 100; i += 2 {

            // 先打印奇数

            fmt.Printf("奇数协程：%d\n", i)

            // 发送信号，通知偶数协程执行（无缓冲通道会阻塞，直到对方接收）

            ch <- struct{}{}

            // 等待偶数协程执行完的信号（再次阻塞，等待对方发送）

            <-ch

        }

        // 奇数打印完毕后，关闭通道，让偶数协程退出循环

        close(ch)

    }()



    // 偶数协程：打印2,4,6...100

    go func() {

        for i := 2; i <= 100; i += 2 {

            // 等待奇数协程的信号（阻塞，直到奇数协程发送）

            _, ok := <-ch

            if !ok { // 通道关闭则退出

                return

            }

            // 打印偶数

            fmt.Printf("偶数协程：%d\n", i)

            // 发送信号，通知奇数协程执行

            ch <- struct{}{}

        }

    }()



    // 主协程等待子协程执行完毕（避免主协程提前退出）

    var wg sync.WaitGroup

    wg.Add(2)

    // 上面两个协程里分别加 wg.Done()，这里简化用for循环阻塞（实际开发用WaitGroup更规范）

    for i := 1; i <= 100; i++ {

        if i%100 == 0 {

            break

        }

    }

}
```



### 用单个channel实现0,1的交替打印



```go
func main() {
	const rounds = 10
	var wg sync.WaitGroup

	// 单个 channel，传递当前需要打印的数字
	// 缓冲为 1，初始放入 0，表示从 0 开始
	ch := make(chan int, 1)
	ch <- 0

	wg.Add(2)

	// 两个协程其实可以合并成一个逻辑，但为了模拟“两个角色”，我们写两个一样的循环
	// 或者更简单地，只用一个协程？不，题目要求两个 goroutine。
	// 如果两个 goroutine 逻辑一样，它们都会尝试读取。
	// 只要通道里有数据，谁抢到谁执行。
	// 执行完后，它把 (当前值 + 1) % 2 放回通道。
	// 这样就能保证交替！因为放回去的值变了，下一个抢到的人打印的就是下一个数。
	
	worker := func(id int) {
		defer wg.Done()
		for {
			val, ok := <-ch
			if !ok {
				return
			}
			
			// 打印
			fmt.Print(val)
			
			// 判断是否结束：如果打印的是 1 且达到了轮数，就关闭通道
			// 这里需要一个共享计数器来判断总次数，或者简单点，打印固定次数后关闭
			// 由于两个协程共享逻辑，我们需要一个原子计数器或者在主协程控制
			
			// 简化版：不在此处判断关闭，由主协程控制次数？
			// 不行，主协程不知道里面打印了多少次。
			// 最好是在通道里传递“剩余次数”或者利用外部原子变量。
			
			// 让我们用一个原子变量来控制总打印次数
			// 但为了保持“单 channel”的纯粹性，我们可以让通道传递结构体 {value, count}
			// 或者，既然题目通常隐含意思是“打印 0,1,0,1... 直到满足条件”，
			// 我们可以让主程序在外部控制，或者让 worker 自己判断。
			
			// 方案：通道传递 {currentVal, remainingSteps}
			// 这样不需要额外的锁或原子变量
		}
	}
	
	// 重新设计数据结构
	type Token struct {
		Value int
		Left  int // 剩余打印次数
	}
	
	// 覆盖之前的 ch 定义
	ch2 := make(chan Token, 1)
	ch2 <- Token{Value: 0, Left: rounds * 2} // 总共打印 20 个数

	workerFunc := func() {
		defer wg.Done()
		for {
			token, ok := <-ch2
			if !ok || token.Left <= 0 {
				return
			}
			
			fmt.Print(token.Value)
			
			// 计算下一个值
			nextVal := 1 - token.Value
			nextLeft := token.Left - 1
			
			if nextLeft > 0 {
				ch2 <- Token{Value: nextVal, Left: nextLeft}
			} else {
				// 任务完成，关闭通道（可选，因为循环会通过 Left<=0 退出，但关闭是好习惯）
				// 注意：如果有多个协程在跑，关闭通道要小心，只能关闭一次。
				// 这里最后一个拿到令牌的人关闭。
				close(ch2)
			}
		}
	}

	go workerFunc()
	go workerFunc()

	wg.Wait()
	fmt.Println()
}
```



### sync.Cond实现多生产者多消费者

```go
package main

import (
	"fmt"
	"math/rand"
	"sync"
	"time"
)

// BoundedBuffer 表示一个有界缓冲区
type BoundedBuffer struct {
	mu    sync.Mutex
	cond  *sync.Cond      // 条件变量
	queue []int           // 底层容器
	cap   int             // 容量上限
}

// NewBoundedBuffer 创建一个新的有界缓冲区
func NewBoundedBuffer(capacity int) *BoundedBuffer {
	bb := &BoundedBuffer{
		queue: make([]int, 0, capacity),
		cap:   capacity,
	}
	// 将条件变量与互斥锁绑定
	bb.cond = sync.NewCond(&bb.mu)
	return bb
}

// Put 生产者方法：放入数据
func (bb *BoundedBuffer) Put(val int) {
	bb.mu.Lock()
	defer bb.mu.Unlock()

	// 1. 如果缓冲区满了，等待
	// 注意：必须使用 for 循环检查条件，防止虚假唤醒
	for len(bb.queue) >= bb.cap {
		bb.cond.Wait()
	}

	// 2. 放入数据
	bb.queue = append(bb.queue, val)
	fmt.Printf("[生产者] 放入: %d, 当前队列长度: %d\n", val, len(bb.queue))

	// 3. 通知消费者：至少有一个数据可取了
	// 在多消费者场景下，Signal 唤醒一个，Broadcast 唤醒所有。
	// 为了简单和安全，这里使用 Signal (因为只要有一个消费者能取就行)
	// 但如果存在多种等待条件，通常建议用 Broadcast。
	bb.cond.Signal() 
}

// Get 消费者方法：取出数据
func (bb *BoundedBuffer) Get() int {
	bb.mu.Lock()
	defer bb.mu.Unlock()

	// 1. 如果缓冲区空了，等待
	for len(bb.queue) == 0 {
		bb.cond.Wait()
	}

	// 2. 取出数据 (FIFO)
	val := bb.queue[0]
	bb.queue = bb.queue[1:]
	fmt.Printf("[消费者] 取出: %d, 当前队列长度: %d\n", val, len(bb.queue))

	// 3. 通知生产者：至少有一个空位了
	bb.cond.Signal()

	return val
}

func main() {
	rand.Seed(time.Now().UnixNano())
	buffer := NewBoundedBuffer(5) // 缓冲区容量为 5
	
	var wg sync.WaitGroup
	done := make(chan struct{})

	// 启动 3 个生产者
	for i := 0; i < 3; i++ {
		wg.Add(1)
		go func(id int) {
			defer wg.Done()
			for j := 0; j < 5; j++ {
				val := id*100 + j
				buffer.Put(val)
				// 模拟生产耗时
				time.Sleep(time.Duration(rand.Intn(200)) * time.Millisecond)
			}
		}(i)
	}

	// 启动 3 个消费者
	for i := 0; i < 3; i++ {
		wg.Add(1)
		go func(id int) {
			defer wg.Done()
			for {
				select {
				case <-done:
					// 尝试消费完剩余数据后退出，或者直接退出
					// 这里为了演示简单，直接返回，实际生产中可能需要消费完剩余项
					return
				default:
					// 非阻塞尝试获取，如果缓冲区空会阻塞在 Get 内部
					// 但由于我们要控制总退出逻辑，这里最好由主程序控制总任务数
					// 为了简化演示，我们让消费者也运行固定次数或通过全局计数控制
					// 这里采用一个简单的策略：消费者一直运行，直到主程序结束
					// 在实际场景中，通常会传递一个总任务计数器
					_ = buffer.Get() 
					time.Sleep(time.Duration(rand.Intn(200)) * time.Millisecond)
				}
			}
		}(i)
	}

	// 等待生产者完成
	// 注意：上面的消费者是死循环，我们需要一种机制告诉消费者停止
	// 更好的方式是统计总生产数量，消费同样数量后停止。
	// 这里为了演示代码简洁性，我们让主程序睡一会，然后强制结束（实际项目请勿这样写）
	// 或者我们可以修改消费者逻辑，让它消费固定总数。
	
	// --- 修正：更严谨的退出机制 ---
	// 假设总共生产 3 * 5 = 15 个任务
	totalTasks := 15
	var consumedCount int
	var countMu sync.Mutex
	
	// 重置 WaitGroup 用于消费者
	// 实际上上面的 goroutine 已经启动了，我们很难动态改变它们的逻辑。
	// 下面的代码仅作为演示“如何运行”，实际退出的完美控制需要结合 atomic 计数器。
	
	// 为了让示例能正常退出且不泄露 goroutine，我们采用最简单的方案：
	// 主线程等待一段时间，确保生产完成，然后关闭程序。
	// 在生产者全部结束后，可以向缓冲区放入特殊标记（如 -1）来通知消费者退出。
	
	// 由于上面的代码已经启动，我们在这里等待生产者逻辑执行完的大致时间
	time.Sleep(3 * time.Second) 
	close(done) // 通知消费者退出
	
	// 等待一小会儿让消费者处理完退出逻辑
	time.Sleep(500 * time.Millisecond)
	
	fmt.Println("程序结束")
}
```
